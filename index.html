plt.figure(figsize=(30,20)) # creates canvas
ax = plt.axes() # get axes


ax.grid(linewidth=0.4, color='#8f8f8f') #grid lines
ax.set_facecolor("black") #set background colour

ax.plot(X,Y,'o-',linewidth=4,markersize=10) #plot
ax.plot(X,D,'r',marker='o',linewidth=4,markersize=10)
ax.plot(X,R,'g',marker='o',linewidth=4,markersize=10)

for i,j in zip(X,Y):# numbers on graph
    ax.annotate(str(j),xy=(i,j+30),color='white',size='15')
for i,j in zip(X,D):# numbers on graph
    ax.annotate(str(j),xy=(i,j-50),color='white',size='15')
for i,j in zip(X,R):# numbers on graph
    ax.annotate(str(j),xy=(i,j+40),color='white',size='15')

#X
ax.set_xlabel('\nDate',size=25,color='#4bb4f2')#label of x
plt.xticks(rotation='vertical',size='20',color='white')#ticks of X
#Y
ax.set_ylabel('Population\n',size=25,color='#4bb4f2')
plt.yticks(size=20,color='white')

plt.tick_params(size=20,color='white')# -
ax.set_title('COVID-19 IN April\n',size=50,color='#28a9ff')

ax.legend(['Confirmed','Deceased','Recovered'],loc="upper left",fontsize='20')
